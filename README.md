## Install

```
$ npm install perfect-square
```


## Usage

```js
const perfectSquare = require('perfect-square')

console.log(perfectSquare(9))
// => true

console.log(perfectSquare(11))
// => false

console.log(perfectSquare(25))
// => true
```
